32# @namespace ___DOC_CMAKE
33# @brief
34# Note they functions defined on CMake, not C/C++.
35#
36# @brief
37# Iterates a directory `prm_TestSourcesDir` and
38# Make a test case for every source.
39#
40# @param prm_LibName
41# Base Library name
42#
43# @param prm_TestSourcesDir
44# A directory where the stand-alone test codes locate. \n
45# Every sources under that directory must be stand-alone, which means it must not depends on another memory, function, etc.
46#
47# @param ...
48# Additional Libraries if you want
49#
50# @see ___DOC_CMAKE::ae2f_TEST
51function(ae2f_CoreTestTent prm_LibName prm_TestSourcesDir)
52 if(ae2f_TEST)
53 if(ae2f_CXX)
54 file(GLOB_RECURSE files "${prm_TestSourcesDir}/*")
55 else()
56 file(GLOB_RECURSE files "${prm_TestSourcesDir}/*.c")
57 endif()
58 list(LENGTH files list_length)
59
60 math(EXPR adjusted_length "${list_length} - 1")
61
62 foreach(i RANGE 0 ${adjusted_length})
63 list(GET files ${i} item)
64 get_filename_component(__NAME ${item} NAME)
65 add_executable("${prm_LibName}-Test-${__NAME}" ${item})
66 target_link_libraries("${prm_LibName}-Test-${__NAME}" ${ARGN} ${prm_LibName})
67 add_test(NAME "${prm_LibName}-Test-${__NAME}" COMMAND "${prm_LibName}-Test-${__NAME}")
68 endforeach()
69 endif()
70endfunction()

72# @brief
73# Makes a Library installable. \n
74# Configuration file could be selected here
75#
76# @param prm_TarName
77# Library name you want.
78#
79# @param prm_TarPrefix
80# [STATIC | SHARED | INTERFACE]
81#
82# @param prm_includeDir
83# The include directory relative to the project CMakeLists.txt
84#
85# @param prm_namespace
86# Namespace (or header root directory after include)
87#
88# @param prm_configpath
89# The path where the input file for Configuration lies.
90#
91# @param ...
92# The sources for the project.
93function(ae2f_CoreLibTentConfigCustom prm_TarName prm_TarPreFix prm_includeDir prm_namespace prm_configpath)
94 # Namespace Package
95 include(GNUInstallDirs)
96
97 include_directories(${prm_includeDir})
98 add_library(${prm_namespace}-${prm_TarName} ${prm_TarPreFix} ${ARGN})
99 add_library(${prm_namespace}::${prm_TarName} ALIAS ${prm_namespace}-${prm_TarName})
100
101 if(${ae2f_nakedalais})
102 add_library(${prm_TarName} ALIAS ${prm_namespace}-${prm_TarName})
103 endif()
104
105 target_include_directories(
106 ${prm_namespace}-${prm_TarName} INTERFACE
107 $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/${prm_includeDir}/>
108 $<INSTALL_INTERFACE:${prm_includeDir}/${prm_namespace}/>
109 )
110
111 # Install Settings
112 install(TARGETS ${prm_namespace}-${prm_TarName}
113 EXPORT ${prm_namespace}-${prm_TarName}Targets
114 LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
115 INCLUDES DESTINATION ${CMAKE_INSTALL_INCLUDEDIR})
116
117 install(DIRECTORY ${prm_includeDir}/${prm_namespace}
118 DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}/${prm_namespace}
119 )
120
121 # Package
122 install(EXPORT ${prm_namespace}-${prm_TarName}Targets
123 FILE ${prm_namespace}-${prm_TarName}Targets.cmake
124 NAMESPACE ae2f::
125 DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/
126 )
127
128 # Pack Conf
129 include(CMakePackageConfigHelpers)
130 configure_package_config_file(
131 ${prm_configpath}
132 ${CMAKE_CURRENT_BINARY_DIR}/${prm_namespace}-${prm_TarName}Config.cmake
133 INSTALL_DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/
134 )
135
136 install(FILES
137 ${CMAKE_CURRENT_BINARY_DIR}/${prm_namespace}-${prm_TarName}Config.cmake
138 DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake
139 )
140
141 set(${prm_namespace}__${prm_TarName}__TENT ${prm_namespace}-${prm_TarName} CACHE STRING ${prm_namespace}-${prm_TarName})
142endfunction()

172# @brief
173# Generate an interface project for document code for cmake utility functions. \n
174# Available when ___DOC_CMAKE::ae2f_DOC is ON.
175#
176# @param prm_TarName
177# Library name you want.
178#
179# @param prm_includeDir
180# Where the documents exist
181# The include directory relative to the project CMakeLists.txt
182#
183# @param prm_namespace
184# Namespace (or header root directory after include)
185#
186# @param ...
187# The past documents name
188# @see ___DOC_CMAKE::ae2f_CoreLibTent
189# @see ___DOC_CMAKE::ae2f_DOC
190function(ae2f_CoreUtilityDocTent prm_TarName prm_includeDir prm_namespace)
191 if(ae2f_DOC)
192 file(GLOB_RECURSE src ${prm_includeDir} "*.cmake.hpp")
193 ae2f_CoreLibTent(${prm_TarName}-CMakeDoc INTERFACE ${prm_includeDir} ${prm_namespace}doc ${src})
194 foreach(lib ${ARGN})
195 target_link_libraries(${prm_TarName}-CMakeDoc INTERFACE ${lib}-CMakeDoc)
196 endforeach()
197 endif()
198endfunction()

200# Domain name customization.
201function(ae2f_CoreLibFetch_DNS prm_AuthorName prm_namespace prm_TarName prm_TagName prm_Domain)
202 if(NOT TARGET ${prm_namespace}::${prm_TarName})
203 if(NOT EXISTS ${ae2f_ProjRoot}/${ae2f_submod}/${prm_AuthorName}/${prm_TarName}/CMakeLists.txt)
204 execute_process(
205 COMMAND
206 git clone
207 https://${prm_Domain}/${prm_AuthorName}/${prm_TarName}
208 ${ae2f_ProjRoot}/${ae2f_submod}/${prm_AuthorName}/${prm_TarName}
209 --branch ${prm_TagName} ${ARGN}
210 RESULT_VARIABLE result
211 )
212
213 if(result)
214 message(FATAL_ERROR "Fetching ${prm_AuthorName}/${prm_TarName} from Github Failed.")
215 endif()
216 endif()
217
218 add_subdirectory(
219 ${ae2f_ProjRoot}/${ae2f_submod}/${prm_AuthorName}/${prm_TarName}
220 ${ae2f_BinRoot}/${ae2f_submod}/${prm_AuthorName}/${prm_TarName}
221 )
222 endif()
223
224 set(
225 ${prm_AuthorName}__${prm_namespace}__${prm_TarName}__FETCHED
226 ${prm_namespace}::${prm_TarName} CACHE STRING ${prm_namespace}::${prm_TarName}
227 )
228
229endfunction()

231# @brief
232# Domain name will stick on github.
233#
234# It will try to fetch the cmake project ae2f-Core like project for Local and Github. \n
235# @see ___DOC_CMAKE::ae2f_LibDirGlob is the given path to check. \n

242# Author name

244# Target name
245# @param prm_TagName
246# Tag name
247function(ae2f_CoreLibFetch_NS prm_AuthorName prm_namespace prm_TarName prm_TagName)

249 ${prm_AuthorName}
250 ${prm_namespace}
251 ${prm_TarName}
252 ${prm_TagName}
253 "github.com"
254 ${ARGN}
255 )
256endfunction()
